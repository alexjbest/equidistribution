\begin{equation*} |\zeta( -\epsilon - it)| = O_\epsilon( |t|^{1/2 + \epsilon}) \end{equation*}
\begin{equation*} |\zeta(\sigma+ it)| \sim | \zeta(1- \sigma-it) \left| \frac{t}{2\pi} \right|^{\frac 12 - \sigma}\text{.} \end{equation*}
\begin{equation*} f(s) = \zeta(s) \zeta(1-s) \end{equation*}

+ Phragmén-Lindelöf, bound it on \(a = -\epsilon, b= 1+ \epsilon\text{:}\)

\begin{equation*} |f(s)| \le |t|^{1/2 +\delta}\text{.} \end{equation*}

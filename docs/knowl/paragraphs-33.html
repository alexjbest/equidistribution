\begin{equation*} \gamma_x^{(l)}\in \mathscr H_d(q) \end{equation*}

\begin{equation*} \sigma (l,d,q) \ll_\epsilon |\mathscr H_d| + d^\epsilon (1+ \frac{d}{q^2 5^{2l}}) \end{equation*}
